#include <stdio.h>
int str_length(char* str)
{
    int l=0;
    while(str[l]!='\0')
    {
        l++;
    }
    return l;
}
int strn_cmp(char *a, char*b, int n)
{
    int count=0;
    for(int i=0;i<n;i++)
    {
        if(a[i]==b[i])
        {
            count++;
            if(count==n)
           {
             return 0;
           }
        }
        if(a[i]>b[i])
        {
           
            return a[i];
        }
        else
        {
            
            return -b[i];
        }
        
    }
}
